REGISTRAZIONE:

DA FE A AUTH:

POST /client/signup

body: {
    email: string,
    password: string,
    name: string
}

res: {
    success: boolean,
    token: string(?)
}

POST /client/login

body: {
    email: string,
    password: string
}

res: {
    success: boolean,
    token: string,
    error: string
}

Campi del token: {
    id: string,
    name: string
}
